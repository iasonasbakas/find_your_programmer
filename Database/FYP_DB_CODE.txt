CREATE TABLE CUSTOMER (
USERNAME VARCHAR(20) NOT NULL,
PASSWORD VARCHAR(30) NOT NULL,
NAME VARCHAR(20) NOT NULL,
SURNAME VARCHAR(20) NOT NULL,
PHONE VARCHAR(13) NOT NULL,
EMAIL VARCHAR(40) NOT NULL,
PRIMARY KEY (USERNAME ));
ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE MEET (
MEETID INT NOT NULL AUTO_INCREMENT,
DATE VARCHAR(10) NOT NULL,
TIME  VARCHAR(5) NOT NULL,
PLACE VARCHAR(50),
EXTRAINFO VARCHAR(200),
PROGRID INT(11) NOT NULL,
USERNAME VARCHAR(20) NOT NULL,
SOCIAL VARCHAR(20),
PRIMARY KEY (MEETID),
CONSTRAINT FKPPROGRID FOREIGN KEY (PROGRID) REFERENCES PROGRAMMER(PROGRID),
CONSTRAINT FKUSERNAME FOREIGN KEY (USERNAME) REFERENCES CUSTOMER(USERNAME));
ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE RATING (
RATID INT NOT NULL AUTO_INCREMENT,
RATING INT NOT NULL,
RATINGINFO VARCHAR(500) NOT NULL,
RATINGDATE VARCHAR(12) NOT NULL,
MEETID INT NOT NULL,
PRIMARY KEY (RATID),
CONSTRAINT FKMEET FOREIGN KEY (MEETID) REFERENCES MEET (MEETID));
ENGINE=InnoDB DEFAULT CHARSET=utf8;
