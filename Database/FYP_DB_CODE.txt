CREATE TABLE PROGRAMMER (
PROGRID INT NOT NULL AUTO_INCREMENT,
NAME CHAR,
SURNAME CHAR,
PHONE INT,
EMAIL VARCHAR(20),
EXTRAINFO VARCHAR(500) NOT NULL,
PRIMARY KEY (PROGRID));

CREATE TABLE PROGRLANG (
LANGID INT NOT NULL AUTO_INCREMENT,
NAME VARCHAR(20),
PRIMARY KEY(LANGID));

CREATE TABLE PROGRSKILLS (
PROGRID INT NOT NULL,
LANGID INT NOT NULL,
CONSTRAINT FKPROGRIDSKILLS FOREIGN KEY (PROGRID) REFERENCES PROGRAMMER(PROGRID),
CONSTRAINT FKLANGIDSKILLS FOREIGN KEY (LANGID) REFERENCES PROGRLANG(LANGID));

CREATE TABLE CUSTOMER (
CUSTID INT NOT NULL AUTO_INCREMENT,
USERNAME VARCHAR(20) NOT NULL,
PASSWORD VARCHAR(30) NOT NULL,
NAME VARCHAR(15) NOT NULL,
SURNAME VARCHAR(15) NOT NULL,
PHONE INT NOT NULL,
EMAIL VARCHAR(20) NOT NULL,
PRIMARY KEY (CUSTID));

CREATE TABLE MEET (
MEETID INT NOT NULL AUTO_INCREMENT,
DATE DATETIME,
PLACE VARCHAR(100),
EXTRAINFO VARCHAR(200),
PROGRID INT,
CUSTID INT,
PRIMARY KEY (MEETID),
CONSTRAINT FKPPROGRID FOREIGN KEY (PROGRID) REFERENCES PROGRAMMER(PROGRID),
CONSTRAINT FKCUSTID FOREIGN KEY (CUSTID) REFERENCES CUSTOMER(CUSTID));

CREATE TABLE RATING (
RATID INT NOT NULL AUTO_INCREMENT,
RATING INT NOT NULL,
RATINGINFO VARCHAR(500) NOT NULL,
RATINGDATE DATETIME NOT NULL,
MEETID INT NOT NULL,
PRIMARY KEY (RATID),
CONSTRAINT FKMEET FOREIGN KEY (MEETID) REFERENCES MEET (MEETID));
